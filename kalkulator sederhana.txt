#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "=== KALKULATOR SEDERHANA ===" << endl;
    cout << "Masukkan angka pertama: ";
    cin >> a;

    cout << "Masukkan operator (+ - * /): ";
    cin >> op;

    cout << "Masukkan angka kedua: ";
    cin >> b;

    cout << "Hasil: ";

    switch(op) {
        case '+':
            cout << a + b;
            break;
        case '-':
            cout << a - b;
            break;
        case '*':
            cout << a * b;
            break;
        case '/':
            if (b != 0)
                cout << a / b;
            else
                cout << "Error! Pembagian dengan nol.";
            break;
        default:
            cout << "Operator tidak valid!";
    }

    return 0;
}
